CREATE TABLE projects (
    id UUID PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    type VARCHAR(50), -- 'video', 'workflow', 'mixed'
    status VARCHAR(50) DEFAULT 'draft',
    config JSONB,
    assets JSONB,
    created_at TIMESTAMP,

    id UUID PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    category VARCHAR(100),
    workflow_definition JSONB,

);
